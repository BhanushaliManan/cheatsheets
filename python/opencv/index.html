  2. OpenCV Basic Boilerplate Boilerplate 
    import cv2

# Different Modes For Images

# -1 = cv2.IMREAD_COLOR = Loads A Color Image, Any Transparency of Image will Be Neglected
# 0 = cv2.IMREAD_GRAYSCALE = Loads Image in Grayscale Mode
# 1 = cv2.IMREAD_UNCHANGED = Loads Image as it is incuding Alpha Channels

mode = 1
img = cv2.imread('assets/img.png', mode) # Default Mode is -1
cv2.imshow('Image', img)
cv2.waitKey(0)
cv2.destroyAllWindows()
  3. Resizing An Image 
    import cv2
img = cv2.imread("assets/img.png", 1)

dimensions = (400, 400) # 400 x 400
img = cv2.resize(img, dimensions)

cv2.imshow("Image", img)
cv2.waitKey(0)
cv2.destroyAllWindows()
  4. Resizing An Image By Relative Values 
    import cv2
img = cv2.imread("assets/img.png", 1)

img = cv2.resize(img, (0, 0), fx=0.5, fy=0.5) # This will Multiply The Image's 'X' Value to 0.5 and 'Y' Value to 0.5, Thus Making it Half The Size Of Whatever It was.

cv2.imshow("Image", img)
cv2.waitKey(0)
cv2.destroyAllWindows()
